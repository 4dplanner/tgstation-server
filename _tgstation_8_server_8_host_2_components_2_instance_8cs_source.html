131  var ddSettingsTask = instanceQuery.Select(x => x.DreamDaemonSettings).Select(x => new DreamDaemonSettings
132  {

135  }).FirstAsync(cancellationToken);

138 
139  using (var repo = await RepositoryManager.LoadRepository(cancellationToken).ConfigureAwait(false))
140  {
141  if (repo == null)
142  return;
143 
144  //start the rev info query
145  var startSha = repo.Head;
146  var revInfoTask = instanceQuery.SelectMany(x => x.RevisionInformations).Where(x => x.CommitSha == startSha).FirstOrDefaultAsync(cancellationToken);
147 
148  //need repo setting to fetch
149  var repositorySettings = await repositorySettingsTask.ConfigureAwait(false);
150  await repo.FetchOrigin(repositorySettings.AccessUser, repositorySettings.AccessToken, null, cancellationToken).ConfigureAwait(false);
151 
152  //take appropriate auto update actions
153  bool shouldSyncTracked;
154  if (repositorySettings.AutoUpdatesKeepTestMerges.Value)
155  {
156  var result = await repo.MergeOrigin(repositorySettings.CommitterName, repositorySettings.CommitterEmail, cancellationToken).ConfigureAwait(false);
157  if (!result.HasValue)
158  return;
159  shouldSyncTracked = result.Value;
160  }
161  else
162  {
163  await repo.ResetToOrigin(cancellationToken).ConfigureAwait(false);
164  shouldSyncTracked = true;
165  }
166 
167  //synch if necessary
168  if (repositorySettings.AutoUpdatesSynchronize.Value && startSha != repo.Head)
169  await repo.Sychronize(repositorySettings.AccessUser, repositorySettings.AccessToken, shouldSyncTracked, cancellationToken).ConfigureAwait(false);
170 
171  //finish other queries
172  var dmSettings = await dmSettingsTask.ConfigureAwait(false);
173  var ddSettings = await ddSettingsTask.ConfigureAwait(false);
174  var revInfo = await revInfoTask.ConfigureAwait(false);
175 
176  //null rev info handling
177  if (revInfo == default)
178  {
179  var currentSha = repo.Head;
180  revInfo = new RevisionInformation
181  {

183  OriginCommitSha = currentSha,
184  Instance = new Models.Instance
185  {

187  }
188  };
189  db.Instances.Attach(revInfo.Instance);
190  }
191 
192  //finally start compile

194  }
195 
196  db.CompileJobs.Add(job);
197  await db.Save(cancellationToken).ConfigureAwait(false);
198  }).ConfigureAwait(false);

201  }
