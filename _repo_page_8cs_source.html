172  private void RepoBGW_DoWork(object sender, DoWorkEventArgs e)

177  switch (action) {

180  break;
181  case RepoAction.Checkout:
182  repoError = Repo.Checkout(CheckoutBranch);
183  break;
184  case RepoAction.Merge:
185  repoError = Repo.Update(false);
186  break;
187  case RepoAction.Reset:
188  repoError = Repo.Reset(true);
189  break;
190  case RepoAction.Test:
191  repoError = Repo.MergePullRequest(TestPR);
192  break;
193  case RepoAction.Update:
194  repoError = Repo.Update(true);
195  break;
196  case RepoAction.Wait:
197  break;
198  case RepoAction.GenCL:
199  var result = Repo.GenerateChangelog(out repoError);
200  if(repoError != null)
201  repoError += ": " + result;
202  break;
203  default:
204  //reeee
205  return;
206  }
207 
208  do
209  {
210  Thread.Sleep(1000);
211  RepoBGW.ReportProgress(Repo.CheckoutProgress());
