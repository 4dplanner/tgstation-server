68  static void GetRepositoryOwnerName(string remote, out string owner, out string name)
69  {
70  //Assume standard gh format: [(git)|(https)]://github.com/owner/repo(.git)[0-1]
71  //Yes use .git twice in case it was weird
72  var toRemove = new string[] { ".git", "/", ".git" };
73  foreach (string item in toRemove)
74  if (remote.EndsWith(item, StringComparison.OrdinalIgnoreCase))
75  remote = remote.Substring(0, remote.LastIndexOf(item, StringComparison.OrdinalIgnoreCase));
76  var splits = remote.Split('/');
77  name = splits[splits.Length - 1];
78  owner = splits[splits.Length - 2].Split('.')[0];
79  }

150  var localBranchName = String.Format(CultureInfo.InvariantCulture, "pull/{0}/headrefs/heads/{1}", pullRequestNumber, prBranchName);
151  Refspec.Add(String.Format(CultureInfo.InvariantCulture, "pull/{0}/head:{1}", pullRequestNumber, prBranchName));

157  await Task.Factory.StartNew(() =>

160  try

164  {
165  Prune = true,
166  OnProgress = (a) => !cancellationToken.IsCancellationRequested,
167  OnTransferProgress = (a) =>
168  {
169  var percentage = 100 * (((float)a.IndexedObjects + a.ReceivedObjects) / (a.TotalObjects * 2));
170  progressReporter?.Invoke((int)percentage);
171  return !cancellationToken.IsCancellationRequested;

175  }

183  }

191  FastForwardStrategy = FastForwardStrategy.NoFastForward,
192  SkipReuc = true
193  });

201  }
202
