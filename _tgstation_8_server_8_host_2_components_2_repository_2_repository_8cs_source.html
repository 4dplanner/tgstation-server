74  static void GetRepositoryOwnerName(string remote, out string owner, out string name)
75  {
76  //Assume standard gh format: [(git)|(https)]://github.com/owner/repo(.git)[0-1]
77  //Yes use .git twice in case it was weird
78  var toRemove = new string[] { ".git", "/", ".git" };
79  foreach (string item in toRemove)
80  if (remote.EndsWith(item, StringComparison.OrdinalIgnoreCase))
81  remote = remote.Substring(0, remote.LastIndexOf(item, StringComparison.OrdinalIgnoreCase));
82  var splits = remote.Split('/');
83  name = splits[splits.Length - 1];
84  owner = splits[splits.Length - 2].Split('.')[0];
85  }
