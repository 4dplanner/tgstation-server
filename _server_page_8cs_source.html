331  List<Task<PullRequest>> pullsRequests = null;

333  //find out which of the PRs have been merged

339  res = await Task.Factory.StartNew(() => repo.Update(true));
340 
341  if(res != null)

352  return;
353  }
354 
355  res = await Task.Factory.StartNew(() => repo.SynchronizePush());
356 
357  if (res != null)
358  {
359  MessageBox.Show(res, "Error synchronizing commits");
360  return;

362 
363  if (pullsRequests != null)
364  Task.WaitAll(pullsRequests.ToArray());
365 
366  foreach (var I in pullsRequests)
367  if (I.Result.Merged)
368  pulls.RemoveAll(x => x.Number == I.Result.Number);
369 
370  var results = new List<string>();

372  retry:
373  await Task.Factory.StartNew(() => res = repo.MergePullRequest(I.Number, I.Sha));
374  if (res != null)
375  switch(MessageBox.Show(res, "Error Re-merging Pull Request", MessageBoxButtons.AbortRetryIgnore))
376  {
377  case DialogResult.Abort:

379  case DialogResult.Retry:

382  }

385  if (res != null)

387  }

389  {
390  UseWaitCursor = false;

392  }
