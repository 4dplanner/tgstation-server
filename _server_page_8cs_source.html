370  private void FullUpdateWorker_DoWork(object sender, DoWorkEventArgs e)
371  {
372  try
373  {
374  var Repo = Interface.GetComponent<ITGRepository>();
375  var DM = Interface.GetComponent<ITGCompiler>();
376  switch (fuAction)
377  {
378  case FullUpdateAction.Testmerge:
379  updateError = Repo.MergePullRequest(testmergePR);
380  if (updateError == null)
381  {
382  Repo.GenerateChangelog(out updateError);
383  updateError = DM.Compile(true) ? updateError : "Compilation failed!";
384  }
385  break;
386  case FullUpdateAction.UpdateHard:
387  updateError = Repo.Update(true);
388  if (updateError == null)
389  {
390  Repo.GenerateChangelog(out updateError);
391  if (updateError == null)
392  updateError = Repo.SynchronizePush();
393  updateError = DM.Compile(true) ? updateError : "Compilation failed!";
394  }
395  break;
396  case FullUpdateAction.UpdateHardTestmerge:
397  updateError = Repo.Update(true);
398  if (updateError == null)
399  {
400  Repo.GenerateChangelog(out updateError);
401  if (updateError == null)
402  updateError = Repo.SynchronizePush();
403  updateError = Repo.MergePullRequest(testmergePR);
404  if (updateError == null)
405  {
406  Repo.GenerateChangelog(out updateError);
407  updateError = DM.Compile(true) ? updateError : "Compilation failed!";
408  }
409  }
410  break;
411  case FullUpdateAction.UpdateMerge:
412  updateError = Repo.Update(false);
413  if (updateError == null)
414  {
415  Repo.GenerateChangelog(out updateError);
416  if (updateError == null)
417  Repo.SynchronizePush(); //not an error 99% of the time if this fails, just a dirty tree
418  updateError = DM.Compile(true) ? updateError : "Compilation failed!";
419  }
420  break;
421  case FullUpdateAction.Reset:
422  updateError = Repo.Reset(true);
423  if (updateError == null)
424  {
425  Repo.GenerateChangelog(out updateError);
426  updateError = DM.Compile(true) ? updateError : "Compilation failed!";
427  }
428  break;
429  }
430  }
431  catch (Exception ex)
432  {
433  Program.ServiceDisconnectException(ex);
434  }
435  }
