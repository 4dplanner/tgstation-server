227  public Task<object> HandleInterop(IQueryCollection query, CancellationToken cancellationToken)
228  {
229  if (query == null)
230  throw new ArgumentNullException(nameof(query));
231 
232  object content = new object();
233  var status = HttpStatusCode.OK;
234 
235  if (query.TryGetValue(InteropConstants.DMParameterCommand, out StringValues values))
236  switch (values.First())
237  {
238  case InteropConstants.DMCommandIdentify:
239  lock (this)
240  if (portClosed)
241  content = new Dictionary<string, int> { { InteropConstants.DMParameterNewPort, nextPort } };
242  break;
243  case InteropConstants.DMCommandOnline:
244  lock (this)
245  if (portClosed)
246  {
247  reattachInformation.Port = nextPort;
248  portAssignmentTcs.SetResult(true);
249  portAssignmentTcs = null;
250  portClosed = false;
251  }
252  break;
253  case InteropConstants.DMCommandApiValidate:
254  apiValidated = true;
255  break;
256  case InteropConstants.DMCommandWorldReboot:
257  if (ClosePortOnReboot)
258  {
259  content = new Dictionary<string, int> { { InteropConstants.DMParameterNewPort, 0 } };
260  portClosed = true;
261  }
262  else
263  ClosePortOnReboot = true;
264  var oldTcs = rebootTcs;
265  rebootTcs = new TaskCompletionSource<object>();
266  oldTcs.SetResult(null);
267  break;
268  default:
269  status = HttpStatusCode.BadRequest;
270  content = new ErrorMessage { Message = "Requested command not supported!" };
271  break;
272  }
273  return Task.FromResult<object>(new { STATUS = status, CONTENT = content });
274  }
