251  public Task HandleInterop(CommCommand command, CancellationToken cancellationToken)
252  {
253  if (command == null)
254  throw new ArgumentNullException(nameof(command));
255 
256  var query = command.Parameters;
257 
258  object content;
259  if (query.TryGetValue(Constants.DMParameterCommand, out var method))
260  {
261  content = new object();
262  switch (method)
263  {
264  case Constants.DMCommandIdentify:
265  lock (this)
266  if (portClosed)
267  content = new Dictionary<string, int> { { Constants.DMParameterData, nextPort } };
268  break;
269  case Constants.DMCommandOnline:
270  lock (this)
271  if (portClosed)
272  {
273  reattachInformation.Port = nextPort;
274  portAssignmentTcs.SetResult(true);
275  portAssignmentTcs = null;
276  portClosed = false;
277  }
278  break;
279  case Constants.DMCommandApiValidate:
280  apiValidated = true;
281  break;
282  case Constants.DMCommandWorldReboot:
283  if (ClosePortOnReboot)
284  {
285  content = new Dictionary<string, int> { { Constants.DMParameterData, 0 } };
286  portClosed = true;
287  }
288  else
289  ClosePortOnReboot = true;
290  var oldTcs = rebootTcs;
291  rebootTcs = new TaskCompletionSource<object>();
292  oldTcs.SetResult(null);
293  break;
294  default:
295  content = new ErrorMessage { Message = "Requested command not supported!" };
296  break;
297  }
298  }
299  else
300  content = new ErrorMessage { Message = "Missing command parameter!" };
301 
302  var json = JsonConvert.SerializeObject(content);
303  return SendCommand(String.Format(CultureInfo.InvariantCulture, "{0}&{1}={2}", byondTopicSender.SanitizeString(Constants.DMTopicInteropResponse), byondTopicSender.SanitizeString(Constants.DMParameterData), byondTopicSender.SanitizeString(json)), cancellationToken);
304  }
