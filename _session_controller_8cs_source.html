264  public async Task HandleInterop(CommCommand command, CancellationToken cancellationToken)
265  {
266  if (command == null)
267  throw new ArgumentNullException(nameof(command));
268 
269  var query = command.Parameters;
270 
271  object content;
272  Action postRespond = null;
273  if (query.TryGetValue(Constants.DMParameterCommand, out var method))
274  {
275  content = new object();
276  switch (method)
277  {
278  case Constants.DMCommandServerPrimed:
279  //currently unused, maybe in the future
280  break;
281  case Constants.DMCommandEndProcess:
282  TerminationWasRequested = true;
283  process.Terminate();
284  return;
285  case Constants.DMCommandNewPort:
286  lock (this)
287  {
288  if (!query.TryGetValue(Constants.DMParameterData, out var stringPort) || !UInt16.TryParse(stringPort, out var currentPort))
289  {
291  logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
292  break;
293  }
294 
295  if (!nextPort.HasValue)
296  //not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
297  reattachInformation.Port = currentPort;
298  else
299  {
300  //nextPort is ready, tell DD to switch to that
301  //if it fails it'll kill itself
302  content = new Dictionary<string, ushort> { { Constants.DMParameterData, nextPort.Value } };
303  reattachInformation.Port = nextPort.Value;
304  nextPort = null;
305 
306  //we'll also get here from SetPort so complete that task
307  var tmpTcs = portAssignmentTcs;
308  portAssignmentTcs = null;
309  if (tmpTcs != null)
310  postRespond = () => tmpTcs.SetResult(true);
311  }
312 
313  portClosedForReboot = false;
314  }
315  break;
316  case Constants.DMCommandApiValidate:
317  apiValidated = true;
318  break;
319  case Constants.DMCommandWorldReboot:
320  if (ClosePortOnReboot)
321  {
322  content = new Dictionary<string, int> { { Constants.DMParameterData, 0 } };
323  portClosedForReboot = true;
324  }
325  var oldTcs = rebootTcs;
326  rebootTcs = new TaskCompletionSource<object>();
327  postRespond = () => oldTcs.SetResult(null);
328  break;
329  default:
330  content = new ErrorMessage { Message = "Requested command not supported!" };
331  break;
332  }
333  }
334  else
335  content = new ErrorMessage { Message = "Missing command parameter!" };
336 
337  var json = JsonConvert.SerializeObject(content);
338  var response = await SendCommand(String.Format(CultureInfo.InvariantCulture, "{0}&{1}={2}", byondTopicSender.SanitizeString(Constants.DMTopicInteropResponse), byondTopicSender.SanitizeString(Constants.DMParameterData), byondTopicSender.SanitizeString(json)), cancellationToken).ConfigureAwait(false);
339 
340  if (response != Constants.DMResponseSuccess)
341  logger.LogWarning("Recieved error response while responding to interop: {0}", response);
342 
343  postRespond?.Invoke();
344  }
