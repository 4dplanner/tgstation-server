193  public Task<object> HandleInterop(IQueryCollection query, CancellationToken cancellationToken)
194  {
195  if (query == null)
196  throw new ArgumentNullException(nameof(query));
197 
198  object content = new object();
199  var status = HttpStatusCode.OK;
200 
201  if (query.TryGetValue(InteropConstants.DMParameterCommand, out StringValues values))
202  switch (values.First())
203  {
204  case InteropConstants.DMCommandIdentify:
205  lock (this)
206  if (portClosed)
207  content = new Dictionary<string, int> { { InteropConstants.DMParameterNewPort, nextPort } };
208  break;
209  case InteropConstants.DMCommandOnline:
210  lock (this)
211  if (portClosed)
212  {
213  reattachInformation.Port = nextPort;
214  portAssignmentTcs.SetResult(true);
215  portAssignmentTcs = null;
216  portClosed = false;
217  }
218  break;
219  case InteropConstants.DMCommandApiValidate:
220  apiValidated = true;
221  break;
222  case InteropConstants.DMCommandWorldReboot:
223  if (ClosePortOnReboot)
224  {
225  content = new Dictionary<string, int> { { InteropConstants.DMParameterNewPort, 0 } };
226  portClosed = true;
227  }
228  else
229  ClosePortOnReboot = true;
230  var oldTcs = rebootTcs;
231  rebootTcs = new TaskCompletionSource<object>();
232  oldTcs.SetResult(null);
233  break;
234  default:
235  status = HttpStatusCode.BadRequest;
236  content = new { message = "Requested command not supported!" };
237  break;
238  }
239  return Task.FromResult<object>(new { STATUS = status, CONTENT = content });
240  }
