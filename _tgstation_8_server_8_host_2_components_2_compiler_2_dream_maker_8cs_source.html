152  {
153  var timeoutTask = Task.Delay(timeoutAt - now, cancellationToken);
154 
155  await Task.WhenAny(controller.Lifetime, timeoutTask).ConfigureAwait(false);
156  cancellationToken.ThrowIfCancellationRequested();
157  }
158 
159  if (controller.Lifetime.IsCompleted)
160  {
161  var validationStatus = controller.ApiValidationStatus;
162  logger.LogTrace("API validation status: {0}", validationStatus);
163  switch (validationStatus)
164  {
165  case ApiValidationStatus.RequiresUltrasafe:
166  job.MinimumSecurityLevel = DreamDaemonSecurity.Ultrasafe;
167  return;
168  case ApiValidationStatus.RequiresSafe:
169  if (securityLevel == DreamDaemonSecurity.Ultrasafe)
170  throw new JobException("This game must be run with at least the 'Safe' DreamDaemon security level!");
171  job.MinimumSecurityLevel = DreamDaemonSecurity.Safe;
172  return;
173  case ApiValidationStatus.RequiresTrusted:
174  if (securityLevel != DreamDaemonSecurity.Trusted)
175  throw new JobException("This game must be run with at least the 'Trusted' DreamDaemon security level!");
176  job.MinimumSecurityLevel = DreamDaemonSecurity.Trusted;
177  return;
178  case ApiValidationStatus.NeverValidated:
179  break;
180  case ApiValidationStatus.BadValidationRequest:
181  throw new JobException("Recieved an unrecognized API validation request from DreamDaemon!");
182  case ApiValidationStatus.UnaskedValidationRequest:
183  default:
184  throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Session controller returned unexpected ApiValidationStatus: {0}", validationStatus));
185  }
186  }
187 
188  throw new JobException("DMAPI validation timed out!");

221  async Task ModifyDme(Models.CompileJob job, CancellationToken cancellationToken)

224  var dmePath = ioManager.ConcatPath(dirA, String.Join('.', job.DmeName, DmeExtension));

226 
227  var dmeModificationsTask = configuration.CopyDMFilesTo(dmePath, ioManager.ResolvePath(dirA), cancellationToken);
228 
229  var dmeBytes = await dmeReadTask.ConfigureAwait(false);
230  var dme = Encoding.UTF8.GetString(dmeBytes);
231 
232  var dmeModifications = await dmeModificationsTask.ConfigureAwait(false);
233 
234  if (dmeModifications == null || dmeModifications.TotalDmeOverwrite)
235  {
236  if (dmeModifications != null)
237  logger.LogDebug(".dme replacement configured!");
238  else
239  logger.LogTrace("No .dme modifications required.");
240  return;
241  }
242 
243  if (dmeModifications.HeadIncludeLine != null)
244  logger.LogDebug("Head .dme include line: {0}", dmeModifications.HeadIncludeLine);
245  if (dmeModifications.TailIncludeLine != null)
246  logger.LogDebug("Tail .dme include line: {0}", dmeModifications.TailIncludeLine);
247 
248  var dmeLines = new List<string>(dme.Split(new[] { Environment.NewLine }, StringSplitOptions.None));
249  for (var I = 0; I < dmeLines.Count; ++I)
250  {
251  var line = dmeLines[I];
252  if (line.Contains("BEGIN_INCLUDE", StringComparison.Ordinal) && dmeModifications.HeadIncludeLine != null)
253  {
254  dmeLines.Insert(I + 1, dmeModifications.HeadIncludeLine);
255  ++I;
256  }
257  else if (line.Contains("END_INCLUDE", StringComparison.Ordinal) && dmeModifications.TailIncludeLine != null)
258  {
259  dmeLines.Insert(I, dmeModifications.TailIncludeLine);
260  break;
261  }
262  }
263 
264  dmeBytes = Encoding.UTF8.GetBytes(String.Join(Environment.NewLine, dmeLines));
265  await ioManager.WriteAllBytes(dmePath, dmeBytes, cancellationToken).ConfigureAwait(false);
