109  {
110  public RepoStatusCommand()
111  {
112  Keyword = "status";
113  }
114  protected override ExitCode Run(IList<string> parameters)
115  {
116  var Repo = Interface.GetComponent<ITGRepository>();
117  var busy = Repo.OperationInProgress();
118  if (!busy)
119  {
120  OutputProc("Repo: Idle");
121  var head = Repo.GetHead(false, out string error);
122  if (head == null)
123  head = "Error: " + error;
124  var remotehead = Repo.GetHead(true, out error);
125  if (remotehead == null)
126  remotehead = "Error: " + error;
127  var branch = Repo.GetBranch(out error);
128  if (branch == null)
129  branch = "Error: " + error;
130  var remote = Repo.GetRemote(out error);
131  if (remote == null)
132  remote = "Error: " + error;
133  OutputProc("Remote: " + remote + " (" + remotehead + ")");
134  OutputProc("Branch: " + branch);
135  OutputProc("HEAD: " + head);
136  OutputProc("Push testmerge commits: " + (Repo.PushTestmergeCommits() ? "ON" : "OFF"));
137  OutputProc(String.Format("Committer Identity: {0} ({1})", Repo.GetCommitterName(), Repo.GetCommitterEmail()));
138  }
139  else
140  {
141  OutputProc("Repo: Busy");
142  var progress = Repo.CheckoutProgress();
143  if (progress != -1)
144  {
145  var eqs = "";
146  for (var I = 0; I < progress / 10; ++I)
147  eqs += "=";
148  var dshs = "";
149  for (var I = 0; I < 10 - (progress / 10); ++I)
150  eqs += "-";
151  OutputProc(String.Format("Progress: [{0}{1}] {2}%", eqs, dshs, progress));
152  }
153  }
154  return ExitCode.Normal;
155  }
156  public override string GetHelpText()
157  {
158  return "Shows the busy status of the repo, remote, branch, and HEAD information";
159  }
